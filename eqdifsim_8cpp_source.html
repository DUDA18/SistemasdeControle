69 void EqdifSim<UsedType>::d2c()
70 {
71  Matrix<UsedType> Mat, E, Temp, Root, I, ZeroVector, IAd;
72 
73  Mat = this->Ad.eigenvalues();
74  IAd.eye(this->Ad.getRows());
75  for(int j = 1; j < 4; j++)//deixar a ordem flexivel
76  E.add( 1, j+1, pow(this->SampleTime, j)/factorial(j));
77 
78  I.eye(E.getCols()-2);
79  ZeroVector.zeros( E.getCols()-2, 1);
80 
81  for (int i = 0; i < Mat.getRows(); i++)
82  {
83  E.add( 1, 1, 1 - Mat(1, i+1));
84  Temp = E||(I|ZeroVector);
85  Root = Temp.eigenvalues();
86  float autovalor = max(abs(Root));
87  this->A.add( i+1, i+1, autovalor);
88  }
